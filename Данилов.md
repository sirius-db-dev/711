
### 1. Вывести модель, категорию, бренд продуктов из категорий смартфоны и ноутбуки брендов Apple и Samsung.
### 2. Вывести названия компаний (НЕ брендов), поставляющие бренды как Apple, так и Samsung.
### 3. Вывести названия компаний, поставляющие часы, но не смартфоны.
### 4. Вывести названия компаний, поставляющих продукцию бренда Apple либо имеющие транспортные средства с объемом(body_volume) от от 27 до 30.
### 5. Вывести минимальную, максимальную, среднюю и суммарную стоимость электронных часов компании Apple.
### 6. Вывести модели мониторов, имеющих максимальную цену в бд.
### 7. Вывести количество различных категорий продуктов в бд.

```sql
-- 1. Вывести модель, категорию, бренд продуктов из категорий смартфоны и ноутбуки брендов Apple и Samsung.
select p.model, p.category, p.brand
from product_qos p
where (p.category = 'смартфоны' or p.category = 'ноутбуки') and (p.brand = 'Apple' or p.brand = 'Samsung')


-- 2. Вывести названия компаний (НЕ брендов), поставляющие бренды как Apple, так и Samsung.
select s.name
from supplier_mh6 s
join supplier_product_info_pcd sp on sp.supplier_id = s.id
join product_qos p on p.id = sp.product_id
where p.brand = 'Apple'
intersect
select s.name
from supplier_mh6 s
join supplier_product_info_pcd sp on sp.supplier_id = s.id
join product_qos p on p.id = sp.product_id
where p.brand = 'Samsung'


-- 3. Вывести названия компаний, поставляющие часы, но не смартфоны.
select s.name
from supplier_mh6 s
join supplier_product_info_pcd sp on sp.supplier_id = s.id
join product_qos p on p.id = sp.product_id
where p.category = 'часы'
except
select s.name
from supplier_mh6 s
join supplier_product_info_pcd sp on sp.supplier_id = s.id
join product_qos p on p.id = sp.product_id
where p.category = 'смартфоны'


-- 4. Вывести названия компаний, поставляющих продукцию бренда Apple либо имеющие транспортные средства с объемом(body_volume) от от 27 до 30.
select distinct s.name
from supplier_mh6 s
join supplier_product_info_pcd sp on sp.supplier_id = s.id
join product_qos p on p.id = sp.product_id
join supplier_car_info_vi8 sc on sc.supplier_id = s.id
join car_kml c on c.id = sc.car_id
where p.brand = 'Apple' or (c.body_volume >= 27 and c.body_volume <= 30)

-- или так

select s.name
from supplier_mh6 s
join supplier_product_info_pcd sp on sp.supplier_id = s.id
join product_qos p on p.id = sp.product_id
where p.brand = 'Apple'
union
select s.name
from supplier_mh6 s
join supplier_car_info_vi8 sc on sc.supplier_id = s.id
join car_kml c on c.id = sc.car_id
where c.body_volume >= 27 and c.body_volume <= 30

-- 5. Вывести минимальную, максимальную, среднюю и суммарную стоимость электронных часов компании Apple.
select
	min(sp.price * sp.quantity) min_cost,
	max(sp.price * sp.quantity) max_cost,
	avg(sp.price * sp.quantity) avg_cost,
	sum(sp.price * sp.quantity) sum_cost
from product_qos p
join supplier_product_info_pcd sp on sp.product_id = p.id
where p.category = 'часы' and p.brand = 'Apple'


-- 6. Вывести модели мониторов, имеющих максимальную цену в бд.
select p.model
from product_qos p
join supplier_product_info_pcd sp on sp.product_id = p.id
where p.category = 'мониторы' and sp.price =
(
	select max(sp.price)
	from supplier_product_info_pcd sp
	join product_qos p on p.id = sp.product_id
	where p.category = 'мониторы'
)


-- 7. Вывести количество различных категорий продуктов в бд.
select count(distinct p.category) 
from product_qos p
```

```sql
(
    id              uuid primary key,
    model           text,
    category        text,
    brand           text,
    score           numeric,
    screen_material text,
    screen_size     int,
    height          int,
    width           int,
    weight          int,

);

(
    supplier_id uuid references supplier_mh6,
    product_id uuid references product_qos,

);

create table car_kml
(
    id                uuid primary key,
    model             text,
    body_volume       int,
    lifting_capacity  int
);

create table supplier_car_info_vi8
(
    supplier_id uuid references supplier_mh6,
    car_id uuid references car_kml,
    quantity int,
    primary key (car_id, supplier_id)
);
```
